Match a Path against the full path as a string 
    import scalax.file.Path
    Path("/tmp/file") match {
      case Path("/tmp/file") => println("it's a match")
      case _ => println("no match")
    }
    Path("/tmp/file") match {
      case Path(stringPath) => println("path as a string is:" + stringPath)
      case _ => println("no match")
    }
